num_chars = len(chars)
unique_chars = len(np.unique(chars))

order = 10
x = []
y = []
for i in range(order,num_chars):
    # grab current key, and previous order keys
    next_key = keys[i]
    prev_keys = np.array(keys[i-order:i])
    
    # collect
    x.append(prev_keys)
    y.append(next_key)
x = np.array(x).T
y = np.array(y)[np.newaxis,:]
