# load in dataset
csvname = datapath + 'universal_regression_samples_0.csv'
# csvname =  datapath + 'noisy_sin_sample.csv'

data = np.loadtxt(csvname,delimiter = ',')
x = data[:-1,:]
y = data[-1:,:] 

degrees = np.arange(1,200)
models = []
for j in degrees:
    # import the v1 library
    mylib9 = nonlib.library_v1.superlearn_setup.Setup(x,y)

    # choose features
    mylib9.choose_features(name = 'polys',degree = j)

    # choose normalizer
    mylib9.choose_normalizer(name = 'none')

    # choose cost
    mylib9.choose_cost(name = 'least_squares')

    # fit an optimization
    mylib9.fit(optimizer = 'newtons method',max_its = 1,epsilon = 0) #10**(-15))
    
    # add model to list
    models.append(mylib9)
    
# load up animator
demo9 = nonlib.run_animators.Visualizer(csvname)

# animate based on the sample weight history
demo9.animate_1d_regression_multiple_runs(models,scatter = 'points',show_history = True)
